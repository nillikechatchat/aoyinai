{{- /* /// Params: /// Resources: Where to search for images /// Image: Image URL or path /// Context: page context /// Returns: /// Resource: Hugo image resource object (may be nil) /// Permalink: Image URL /// Height: Image height /// Width: Image width */ -}} {{- $result := dict -}} {{- $image := .Image -}} {{- if $image -}} {{- $url := urls.Parse $image -}} {{- $cleanPath := strings.TrimPrefix "/" $image -}} {{- $permalink := printf "/%s" $cleanPath -}} {{- $resource := false -}} {{- $local := true -}} {{- if not (in (slice "https" "http") $url.Scheme) -}} {{- /* Local image: try page resources first, then static */ -}} {{- $canUseResources := false -}} {{- if .Resources -}} {{- $rType := printf "%T" .Resources -}} {{- if eq $rType "*resources.resourceResources" -}} {{- $canUseResources = true -}} {{- end -}} {{- end -}} {{- if $canUseResources -}} {{- $tryResult := false -}} {{- with try (.Resources.Get $cleanPath) -}} {{- if and (not .Err) .Value -}} {{- $resource = .Value -}} {{- end -}} {{- end -}} {{- end -}} {{- if not $resource -}} {{- $staticResource := resources.Get $cleanPath -}} {{- if $staticResource -}} {{- $resource = $staticResource -}} {{- end -}} {{- end -}} {{- else -}} {{- /* Remote image */ -}} {{- with try (resources.GetRemote $url (dict "timeout" .Context.Site.Params.imageProcessing.external.timeout)) -}} {{- with .Err -}} {{- warnf "Failed to fetch remote resource %q: %s" $url . -}} {{- else with .Value -}} {{- $resource = . -}} {{- $local = false -}} {{- $permalink = $url.String -}} {{- else -}} {{- warnf "Unable to get remote resource %q" $url -}} {{- end -}} {{- end -}} {{- end -}} {{- if $resource -}} {{- if and .Context.Site.Params.imageProcessing.autoOrient (reflect.IsImageResourceProcessable $resource) -}} {{- $filter := images.AutoOrient -}} {{- $resource = $resource | images.Filter $filter -}} {{- end -}} {{- if $local -}} {{- $permalink = $resource.RelPermalink -}} {{- end -}} {{- if reflect.IsImageResourceWithMeta $resource -}} {{- $result = dict "Resource" $resource "Permalink" $permalink "Height" $resource.Height "Width" $resource.Width -}} {{- else -}} {{- $result = dict "Resource" $resource "Permalink" $permalink "Height" nil "Width" nil -}} {{- end -}} {{- else -}} {{- /* Fallback: use cleaned absolute path */ -}} {{- $result = dict "Resource" nil "Permalink" $permalink "Height" nil "Width" nil -}} {{- end -}} {{- end -}} {{- return $result -}}
